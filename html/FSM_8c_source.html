159 bool FSM_IdleTrigger(){
160  // check if we are on a floor with orders
161  // if not -> Update floor_level to floor pluss/minus 0.5
162  assert(m_elevator_variables.direction != ElevatorDirectionNeutral);
163  int elevio_floor = elevio_floorSensor();
164 
165  if (elevio_floor != NO_FLOOR){ // if on floor
166  elevio_floorIndicator(elevio_floor);
167  if ((m_elevator_variables.direction == ElevatorDirectionUp
168  && (m_elevator_buttons[BUTTON_HALL_UP][elevio_floor]
169  || !FSM_orderAbovefloor(elevio_floor)))
170 
171  || (m_elevator_buttons[BUTTON_CAB][elevio_floor])
172 
173  || (m_elevator_variables.direction == ElevatorDirectionDown
174  && (m_elevator_buttons[BUTTON_HALL_DOWN][elevio_floor]
175  || !FSM_orderBelowfloor(elevio_floor))))
176  {
177  // if on floor with order with same direction as elevator or furthest order in that diretion
178  m_elevator_variables.floor_level = (double)elevio_floor;
179  return true;
180  }
181 
182  // if condition above are not met
183  if (m_elevator_variables.direction == ElevatorDirectionUp){
184 
185  m_elevator_variables.floor_level = (double)elevio_floor + 0.5;
186 
187  } else { // direction == down
188  m_elevator_variables.floor_level = (double)elevio_floor - 0.5;
189  }
190  }
191  return false;
192 }

240  int decision = NO_DECISION;
241  switch (m_elevator_variables.direction){
242  case ElevatorDirectionNeutral:
243  for (int floor = 0; floor < N_FLOORS; floor++){
244  for (int button = 0; button < N_BUTTONS; button++){
245  if (m_elevator_buttons[button][floor]){
246 
247  if (m_elevator_variables.floor_level < (double)floor){
248  return SHOULD_UP;
249 
250  } else if (m_elevator_variables.floor_level == (double)floor){
251  decision = SHOULD_STAY;
252 
253  } else{ // m_elevator_variables.floor_level > (double) floor)
254  return SHOULD_DOWN;
255  }
256  }
257  }
258  }
259  break;
260  case ElevatorDirectionUp:
261  for (int floor = (int)m_elevator_variables.floor_level + 1; floor < N_FLOORS; floor++){
262  for (int button = 0; button < N_BUTTONS; button++){
263  if (m_elevator_buttons[button][floor] && (button == BUTTON_CAB || button == BUTTON_HALL_UP || floor == (N_FLOORS - 1))){
264  decision = SHOULD_UP;
265  }
266  }
267  }
268  break;

271  for (int button = 0; button < N_BUTTONS; button++){
